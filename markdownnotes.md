## 1.C代码



```C
# include <stdio.h>
# include <math.h>

int main(void)
{
    long n, i, l, r;
    scanf("%ld", &n);
    long a[n];
    for (i = 0; i < n; ++i)
    {
        scanf("%ld", &a[i]);
    }
    for (i = 0; i < n; ++i)
    {
        l = 0;
        r = 0;
        while ((l+r)*(r-l+1)/2 != a[i])
        {
            if ((l+r)*(r-l+1)/2 < a[i])
                r++;
            else if((l+r)*(r-l+1)/2 > a[i])
                l--;
            else
                break;
        }
        printf("%ld %ld\n", l, r);
    }

    return 0;
}
```
